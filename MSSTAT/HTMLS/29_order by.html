Order By Statement (Examples)


 1. Sorting by a single column (Ascending Order) 

	SELECT EmployeeID, FirstName, LastName, Salary

	FROM Employees

	ORDER BY Salary ASC;

 2. Sorting by a single column (Descending Order) 

	SELECT EmployeeID, FirstName, LastName, Salary 

	FROM Employees

	ORDER BY Salary DESC;

 3. Sorting by multiple columns 

	SELECT EmployeeID, FirstName, LastName, Salary

	FROM Employees

	ORDER BY LastName ASC, FirstName ASC;

 1. Sorting by EmployeeID in Descending Order 

	SELECT EmployeeID, FirstName, LastName, Salary

	FROM Employees

	ORDER BY EmployeeID DESC;

 2. Sorting by FirstName and Salary in Ascending Order 

	SELECT EmployeeID, FirstName, LastName, Salary

	FROM Employees

	ORDER BY FirstName ASC, Salary ASC;

 3. Sorting by Salary in Descending Order with NULL Values Last 

	SELECT EmployeeID, FirstName, LastName, Salary

	FROM Employees

	ORDER BY Salary DESC NULLS LAST;

 4. Sorting by LastName and FirstName (Case-insensitive) 

	SELECT EmployeeID, FirstName, LastName, Salary

	FROM Employees

	ORDER BY LastName COLLATE SQL_Latin1_General_CP1_CI_AS ASC, FirstName COLLATE SQL_Latin1_General_CP1_CI_AS ASC;

 5. Sorting by Salary and FirstName (Descending for Salary, Ascending for FirstName) 

	SELECT EmployeeID, FirstName, LastName, Salary

	FROM Employees

	ORDER BY Salary DESC, FirstName ASC;

 6. Sorting by LastName in Descending Order, and then EmployeeID in Ascending Order 

	SELECT EmployeeID, FirstName, LastName, Salary

	FROM Employees

	ORDER BY LastName DESC, EmployeeID ASC;

 7. Sorting by Salary and then by EmployeeID 

	SELECT EmployeeID, FirstName, LastName, Salary

	FROM Employees

	ORDER BY Salary ASC, EmployeeID ASC;

 8. Sorting by FirstName, LastName, and Salary (All Ascending) 

	SELECT EmployeeID, FirstName, LastName, Salary

	FROM Employees

	ORDER BY FirstName ASC, LastName ASC, Salary ASC;

 9. Sorting by Salary with a Limit of Top 5 Highest Paid Employees 

	SELECT TOP 5 EmployeeID, FirstName, LastName, Salary

	FROM Employees

	ORDER BY Salary DESC;

 10. Sorting by Salary and Filtering for Employees with Salary Greater than a Specific Amount 

	SELECT EmployeeID, FirstName, LastName, Salary

	FROM Employees

	WHERE Salary > 50000

	ORDER BY Salary DESC;
