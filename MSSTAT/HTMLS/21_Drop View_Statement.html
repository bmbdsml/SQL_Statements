Drop View Statement (Examples)


1 < Drop View | If View Exists > 

	 DROP VIEW IF EXISTS dbo.MyView;

2 < Drop View | Check View Object > 

	IF OBJECT_ID('dbo.ProductDetails', 'V') IS NOT NULL

	    DROP VIEW dbo.ProductDetails;

3 < Drop View | Against Given Schema > 

	DROP VIEW IF EXISTS sales.SalesReport;

 4 < Drop View | If View Exists Drop / Create > 

	-- Drop old view before creating a new one

		DROP VIEW IF EXISTS dbo.OrdersSummary;

		-- Create a new view

		CREATE VIEW dbo.OrdersSummary AS
		
			SELECT OrderID, OrderDate, TotalAmount FROM Orders WHERE Status = 'Completed';
