Insert Into Select (Examples)

 1. All Rows 

	INSERT INTO customers (customer_id, first_name, last_name, email)

	SELECT client_id, first_name, last_name, email_address

	FROM client_info;


 2. For Specific Date 

	INSERT INTO employees (employee_id, first_name, last_name, hire_date)

	SELECT employee_id, first_name, last_name, hire_date

	FROM new_employees

	WHERE hire_date > '2024-01-01';


 3. With Concat  

	INSERT INTO employees (employee_id, full_name, hire_date)

	SELECT employee_id, CONCAT(first_name, ' ', last_name), hire_date

	FROM new_employees;

 5. For Status = 'Completed' 

	INSERT INTO completed_orders (order_id, customer_id, order_date, status)

	SELECT order_id, customer_id, order_date, status

	FROM orders

	WHERE status = 'Completed';

 6. With Joins 

	INSERT INTO employee_department (employee_id, department_name)

	SELECT e.employee_id, d.department_name

	FROM employees e

	INNER JOIN departments d ON e.department_id = d.department_id;

 8. Summed Data 

	INSERT INTO monthly_sales_summary (month, total_sales)

	SELECT MONTH(sale_date), SUM(amount)

	FROM sales_data

	GROUP BY MONTH(sale_date);
