Select Into (Examples)


 1-> Syntax 

	SELECT column1, column2, ... INTO new_table_name FROM existing_table WHERE condition;

 2-> Basic Example: Select All Data from One Table Into Another 

	SELECT * INTO NewTable FROM ExistingTable;

 3-> Select Specific Columns and Create a New Table 

	SELECT ProductID, ProductName, Price INTO ProductsCopy FROM Products WHERE Price > 100;

 4-> Select Distinct Records and Create a New Table 

	SELECT DISTINCT CustomerID INTO UniqueCustomers FROM Orders;

 5-> Select Data with Aggregation 

	SELECT Department, COUNT(EmployeeID) AS EmployeeCount
	
	INTO DepartmentEmployeeCount
	
	FROM Employees
	
	GROUP BY Department;

 6-> Create a Backup of a Table 

	SELECT * INTO EmployeesBackup FROM Employees;

 7-> Select Data from Multiple Tables Using JOIN 

	SELECT E.EmployeeID, E.Name, D.DepartmentName

	INTO EmployeeDepartmentInfo

	FROM Employees E

	INNER JOIN Departments D ON E.DepartmentID = D.DepartmentID;

 8-> Select Data with a Complex Condition 

	SELECT CustomerID, OrderID, OrderDate
	
	INTO HighValueOrders
	
	FROM Orders
	
	WHERE OrderAmount > 500 AND OrderDate >= '2025-01-01';

 9-> Select Data from a Query with Subquery 

	SELECT ProductID, ProductName, Price
	
	INTO ExpensiveProducts
	
	FROM Products
	
	WHERE Price > (SELECT AVG(Price) FROM Products);

 10-> Using SELECT INTO with ORDER BY 

	SELECT EmployeeID, Name, Salary
	
	INTO SortedEmployees
	
	FROM Employees
	
	ORDER BY Salary DESC;

 11-> Select Data and Create Table with Modified Column Names 

	SELECT EmployeeID AS EmpID, Name AS FullName, Department AS Dept
	
	INTO EmployeeDetails
	
	FROM Employees;
