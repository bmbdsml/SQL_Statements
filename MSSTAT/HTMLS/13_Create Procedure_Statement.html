Create Procedure (Examples)


1 < GetEmployeeDetails > 

    CREATE PROCEDURE GetEmployeeDetails

        @EmployeeID INT

	AS

	BEGIN
	    -- Select employee details from the Employee table
     SELECT 

        EmployeeID,FirstName,LastName,JobTitle,HireDate FROM Employees WHERE EmployeeID = @EmployeeID;

     END;

     EXEC GetEmployeeDetails @EmployeeID = 5;

2 < InsertEmployee > 

	CREATE PROCEDURE InsertEmployee

	    @FirstName NVARCHAR(50),@LastName NVARCHAR(50),
	    @JobTitle NVARCHAR(50),@HireDate DATE

	AS

	BEGIN

			-- Insert a new record into the Employees table

	    INSERT INTO Employees (FirstName, LastName, JobTitle, HireDate)

	    VALUES (@FirstName, @LastName, @JobTitle, @HireDate);
	END;

	EXEC InsertEmployee @FirstName = 'John', @LastName = 'Doe',
	                    @JobTitle = 'Software Developer',@HireDate = '2025-02-06';

 3 < UpdateEmployee > 

	CREATE PROCEDURE UpdateEmployee

	    @EmployeeID INT,
	    @FirstName NVARCHAR(50),

	    @LastName NVARCHAR(50),
	    @JobTitle NVARCHAR(50),

	    @HireDate DATE

	AS

	BEGIN

				-- Update an existing employee's details

	    UPDATE Employees

	SET 
        	FirstName = @FirstName,
	        LastName = @LastName,
	        JobTitle = @JobTitle,
	        HireDate = @HireDate

	    WHERE EmployeeID = @EmployeeID;

	END;
