Drop Default Constraint Statement (Examples)


1 < Drop Default Constraint > 

	SELECT name FROM sys.default_constraints 

	WHERE parent_object_id = OBJECT_ID('YourTableName') AND col_name(parent_object_id, parent_column_id) = 'YourColumnName';

	ALTER TABLE YourTableName DROP CONSTRAINT YourConstraintName;

	ALTER TABLE Employees DROP CONSTRAINT DF_Employees_Salary;
