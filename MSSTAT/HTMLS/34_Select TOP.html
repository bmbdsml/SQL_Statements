Top Statement (Examples)

 1-> Select Top 5 rows from a table 

	SELECT TOP 5 * FROM Employees;

 2-> Select Top 10 rows ordered by a specific column 

	SELECT TOP 10 * 
	FROM Employees
	ORDER BY Salary DESC;

 3-> Select Top N percentage of rows 

	SELECT TOP 20 PERCENT * 
	FROM Employees 
	ORDER BY Salary DESC;

 4-> Select Top 3 rows with specific column filtering 

	SELECT TOP 3 FirstName, LastName, Salary 
	FROM Employees
	WHERE Department = 'HR';

 5-> Select Top 10 rows with specific sorting and filtering 

	SELECT TOP 10 * 
	FROM Employees
	WHERE Age > 30
	ORDER BY HireDate DESC;

 6-> Select Top 5 rows from each group using ROW_NUMBER() 

	WITH RankedEmployees AS (
	    SELECT FirstName, LastName, Department, Salary,
	           ROW_NUMBER() OVER (PARTITION BY Department ORDER BY Salary DESC) AS RowNum
	    FROM Employees
	)

	SELECT FirstName, LastName, Department, Salary

	FROM RankedEmployees

	WHERE RowNum <= 5

 7-> Select Top 10 rows from a large result set 

	SELECT TOP 10 * 
	FROM (SELECT * FROM Employees WHERE Salary > 50000) AS HighSalaryEmployees
	ORDER BY HireDate DESC;

 8-> Select the first 5 rows from a specific time range 

	SELECT TOP 5 OrderID, OrderDate, CustomerID
	FROM Orders
	WHERE OrderDate BETWEEN '2024-01-01' AND '2024-12-31'
	ORDER BY OrderDate;
