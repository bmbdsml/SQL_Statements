Delete Statement (Examples)


 1 < All Rows > 

	DELETE FROM Employees;

2 < Delete Based on Where Condition > 

	DELETE FROM Employees WHERE DATEDIFF(YEAR, HireDate, GETDATE()) > 5;

3 < Delete Based on | Sub Query > 

	DELETE FROM Orders WHERE CustomerID IN (SELECT CustomerID FROM Customers WHERE Country = 'USA');

4 < Delete Based on | JOIN > 

	DELETE  FROM Employees e

	INNER JOIN Departments d ON e.DepartmentID = d.DepartmentID  WHERE d.Status = 'Closed';

5 < Delete Based on | LIKE > 

	DELETE FROM Employees WHERE LastName LIKE 'S%';

6 < Delete Based on | AND or NOT > 

	DELETE FROM Customers WHERE (Country = 'Canada' AND Age < 18) 

			            OR CustomerID NOT IN (SELECT CustomerID FROM Orders);

7 < Delete Based on | Top N Rows > 

	DELETE TOP (5) FROM Employees WHERE Department = 'Sales';

8 < Delete Based on | Date Range  > 

	DELETE FROM Transactions WHERE TransactionDate < DATEADD(YEAR, -1, GETDATE());

9 < Delete Based on | Exists > 

	DELETE FROM Customers WHERE NOT EXISTS (

	    SELECT 1 FROM Orders WHERE Orders.CustomerID = Customers.CustomerID

			               AND OrderDate > DATEADD(MONTH, -6, GETDATE())

					       );

10 < Delete Based on | Null > 

	DELETE FROM Customers WHERE Email IS NULL;

11 < Delete Based on | Transaction > 

	BEGIN TRANSACTION;

		DELETE FROM Employees WHERE EmployeeID = 123;

			-- COMMIT if everything is okay, or ROLLBACK if there is an issue

	COMMIT;
