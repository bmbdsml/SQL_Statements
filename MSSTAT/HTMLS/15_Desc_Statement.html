Desc Statement (Examples)

1 < DESC | Single Column >

	SELECT EmployeeID, FirstName, LastName, Salary FROM Employees 

	ORDER BY Salary DESC;

2 < DESC | Multiple Column >

	SELECT EmployeeID, FirstName, LastName, Salary FROM Employees 
	
	ORDER BY Salary DESC, LastName ASC;

3 < DESC | Date Column >

	SELECT OrderID, CustomerID, OrderDate FROM Orders 
	
	ORDER BY OrderDate DESC;

4 < DESC | Numeric Column >

	SELECT TOP 5 EmployeeID, FirstName, LastName, Salary FROM Employees 
	
	ORDER BY Salary DESC;

5 < DESC | in Group By Clause >

	SELECT EmployeeID, SUM(SaleAmount) AS TotalSales FROM Sales GROUP BY EmployeeID HAVING SUM(SaleAmount) > 1000

	ORDER BY TotalSales DESC;

6 < DESC | Multiple Column >

	SELECT EmployeeID, DepartmentID, FirstName, LastName, Salary FROM Employees 
	
	ORDER BY DepartmentID DESC, Salary DESC;
