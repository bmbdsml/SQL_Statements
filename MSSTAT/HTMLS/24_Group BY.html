Group By Statement (Examples)


 1. Group BY | Single Column  

		SELECT ProductID, 

			SUM(Quantity) AS TotalQuantity,
	   
			SUM(SaleAmount) AS TotalSales

		FROM Sales

		GROUP BY ProductID ORDER BY ProductID;

 2. Group BY | Multiple Column  

		SELECT ProductID, 

			SaleDate,

			SUM(SaleAmount) AS TotalSales

			FROM Sales

			GROUP BY ProductID, SaleDate

		ORDER BY SaleDate, ProductID;

 3. Group BY | Count the Number of Transactions Per Product   

		SELECT ProductID,
		
			COUNT(SaleID) AS NumberOfTransactions
	   
		FROM  Sales

		GROUP BY  ProductID

		ORDER BY  NumberOfTransactions DESC;

 4. Group BY | Find the Average Sale Amount Per Product   

		 SELECT ProductID,

			AVG(SaleAmount) AS AverageSaleAmount
	   
			FROM Sales

			GROUP BY ProductID

			ORDER BY AverageSaleAmount DESC;

 5. Group BY | HAVING Clause  

		SELECT ProductID,

			SUM(Quantity) AS TotalQuantity
	   
			FROM Sales

			GROUP BY ProductID

			HAVING SUM(Quantity) > 100

			ORDER BY TotalQuantity DESC;

6. Group BY | Year and Month  

		SELECT YEAR(SaleDate) AS SaleYear,

			MONTH(SaleDate) AS SaleMonth,
	   
			SUM(SaleAmount) AS TotalSales
	   
		FROM Sales

		GROUP BY YEAR(SaleDate), MONTH(SaleDate)

		ORDER BY SaleYear DESC, SaleMonth DESC;

7. Group BY | Calculated Field  

		SELECT ProductID,

			YEAR(SaleDate) AS SaleYear,
	   
			SUM(SaleAmount) AS TotalSales
	   
		FROM Sales

		GROUP BY ProductID, YEAR(SaleDate)

		ORDER BY SaleYear DESC, TotalSales DESC;
