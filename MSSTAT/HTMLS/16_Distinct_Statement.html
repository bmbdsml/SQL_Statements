Distinct Statement (Examples)


1 < Selecting distinct values from a single column >

	SELECT DISTINCT Country FROM Customers;

2  < Selecting distinct values from multiple columns >

	SELECT DISTINCT FirstName, LastName FROM Employees;
	
3  < Using DISTINCT with aggregate functions >

	SELECT DISTINCT Department FROM Employees WHERE Salary > 50000;

4  < DISTINCT with ORDER BY >

	SELECT DISTINCT City FROM Orders ORDER BY City;

5  < DISTINCT | JOIN statement > 

	SELECT DISTINCT o.OrderID, c.CustomerName FROM Orders o 
	
	INNER JOIN Customers c ON o.CustomerID = c.CustomerID;

6  < DISTINCT | Where condition > 

	SELECT DISTINCT ProductName FROM Products WHERE CategoryID = 3;

7  < DISTINCT | Sub Query > 

	SELECT CustomerID, (SELECT DISTINCT ProductName FROM OrderDetails WHERE OrderID = Orders.OrderID)
	FROM Orders;

8  < DISTINCT | Group By clause > 

	SELECT DISTINCT Department FROM Employees GROUP BY Department;

9  < DISTINCT | Date column>

	SELECT DISTINCT OrderDate FROM Orders WHERE OrderDate > '2024-01-01';

10  < DISTINCT | Agregate >

	SELECT DISTINCT COUNT(DISTINCT Country) FROM Customers;

11  < DISTINCT | Agregate + Top Clause >

	SELECT DISTINCT TOP 5 City FROM Customers ORDER BY City;
