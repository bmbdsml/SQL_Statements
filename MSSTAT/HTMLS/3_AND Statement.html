AND Statement (Examples)


  1. AND Example-1 

	 SELECT  EmployeeID, EmployeeName, Age, Department  FROM  Employees 
     WHERE  Age > 30  AND  Department = 'Sales';   


  2. AND Example-2  

	 SELECT  EmployeeID, EmployeeName, Age, Department, Salary  FROM  Employees 
     WHERE  Age > 30  AND  Department = 'Sales'  AND  Salary > 50000;

  3. AND Example-3 

	 SELECT  EmployeeID, EmployeeName, Age, Department, Salary  FROM  Employees 
	 WHERE  (Age > 30 OR Department = 'Sales')  AND  Salary > 50000;
